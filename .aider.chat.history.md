import win32gui
import time

class GameDetector:
    def __init__(self, config):
        self.config = config
        self.active_window_title = None

    def detect_active_game(self):
        while True:
            self.active_window_title = win32gui.GetWindowText(win32gui.GetForegroundWindow())
            if any(method in self.config.game_detection_methods for method in ['process_name', 'active_window_title']):
                break
            time.sleep(0.1)

import math

class Analyzer:
    def __init__(self):
        pass

    def analyze_movement(self, mouse_dx_dy):
        flick_intensity = math.sqrt(mouse_dx_dy[0]**2 + mouse_dx_dy[1]**2)
        smoothness = 1 if flick_intensity < 5 else flick_intensity / 5
        micro_correction_frequency = 0
        overflick = flick_intensity > 10
        underflick = flick_intensity < 3
        stability = 1 if flick_intensity < 10 else flick_intensity / 10
        jitter = flick_intensity / 20

        return {
            'flick_intensity': flick_intensity,
            'smoothness': smoothness,
            'micro_correction_frequency': micro_correction_frequency,
            'overflick': overflick,
            'underflick': underflick,
            'stability': stability,
            'jitter': jitter
        }

# Aim/Mouse Behavior Analysis

## Overview

This is a Windows Python desktop app designed for analyzing aim/mouse behavior in games without affecting gameplay or reading game memory.

### Core Requirements

- **No Memory Reading**: The app does not read game memory.
- **No Process Injection**: It does not inject into the game process.
- **No Hooks**: It does not hook into the game.
- **No Input Emulation**: It does not emulate mouse/keyboard input.
- **No Packet Interception**: It does not intercept packets.
- **No Gameplay Interaction**: It runs as a completely separate process and only uses safe standard Windows APIs for passive input monitoring and active-window/process detection.

### Main Goal

The app is a local post-session analysis tool that records mouse behavior safely, detects the selected game or genre, stores session data, analyzes movement patterns, and shows recommendations after the session ends.

### Project Structure

- **app.py**: The main entry point of the application.
- **config.py**: Configuration settings for the app.
- **collector.py**: Collects mouse movement, clicks, timestamps, and active window/process information.
- **game_detector.py**: Detects the active game based on process name or active window title.

- **recommendations.py**: Generates recommendations based on analysis results.

  - **settings_panel.py**: Provides settings for the app.

import win32gui
import time

class GameDetector:
    def __init__(self, config):
        self.config = config
        self.active_window_title = None

    def detect_active_game(self):
        while True:
            self.active_window_title = win32gui.GetWindowText(win32gui.GetForegroundWindow())
            if any(method in self.config.game_detection_methods for method in ['process_name', 'active_window_title']):
                break
            time.sleep(0.1)

import math

class Analyzer:
    def __init__(self):
        pass

    def analyze_movement(self, mouse_dx_dy):
        flick_intensity = math.sqrt(mouse_dx_dy[0]**2 + mouse_dx_dy[1]**2)
        smoothness = 1 if flick_intensity < 5 else flick_intensity / 5
        micro_correction_frequency = 0
        overflick = flick_intensity > 10
        underflick = flick_intensity < 3
        stability = 1 if flick_intensity < 10 else flick_intensity / 10
        jitter = flick_intensity / 20

        return {
            'flick_intensity': flick_intensity,
            'smoothness': smoothness,
            'micro_correction_frequency': micro_correction_frequency,
            'overflick': overflick,
            'underflick': underflick,
            'stability': stability,
            'jitter': jitter
        }

import json

class Storage:
    def __init__(self, filename):
        self.filename = filename

    def save_session_data(self, session_data):
        with open(self.filename, 'w') as file:
            json.dump(session_data, file)

    def load_session_data(self):
        try:
            with open(self.filename, 'r') as file:
                return json.load(file)
        except FileNotFoundError:
            return {}

# Aim/Mouse Behavior Analysis

## Overview

This is a Windows Python desktop app designed for analyzing aim/mouse behavior in games without affecting gameplay or reading game memory.

### Core Requirements

- **No Memory Reading**: The app does not read game memory.
- **No Process Injection**: It does not inject into the game process.
- **No Hooks**: It does not hook into the game.
- **No Input Emulation**: It does not emulate mouse/keyboard input.
- **No Packet Interception**: It does not intercept packets.
- **No Gameplay Interaction**: It runs as a completely separate process and only uses safe standard Windows APIs for passive input monitoring and active-window/process detection.

### Main Goal

The app is a local post-session analysis tool that records mouse behavior safely, detects the selected game or genre, stores session data, analyzes movement patterns, and shows recommendations after the session ends.

### Project Structure

- **app.py**: The main entry point of the application.
- **config.py**: Configuration settings for the app.
- **collector.py**: Collects mouse movement, clicks, timestamps, and active window/process information.
- **game_detector.py**: Detects the active game based on process name or active window title.
- **analyzer.py**: Analyzes mouse movement data to calculate flick intensity, smoothness, etc.
- **recommendations.py**: Generates recommendations based on analysis results.
- **storage.py**: Stores session data locally in JSON format.
- **ui/**: Contains the user interface files.
  - **main_window.py**: The main window of the application.
  - **game_selection.py**: Allows users to select a game or genre.
  - **dashboard.py**: Displays session statistics and recommendations.
  - **settings_panel.py**: Provides settings for the app.

### Game/Genre Selection

- Supports both genre mode and specific game mode.
- Genre options: Tactical FPS, Hero / Arena FPS, Battle Royale, Universal.
- Specific game options: CS2, Valorant, Apex Legends, Overwatch 2, Universal.

### Safe Game Detection

- Uses process name and active window title for detection.
- Does not use memory reading, process injection, DLL injection, packet inspection, anti-cheat bypass techniques, or direct game hooks.

### Data Collection

- Collects only safe local session data:
  - Timestamp
  - Mouse dx/dy
  - Optional cursor position if safe and easy to obtain
  - Left/right click events
  - Wheel events if available
  - Session start/end
  - Selected game/genre
  - Detected active process/window title for logging only

- Does not collect:
  - In-game entity data
  - Enemy positions
  - Crosshair data from the game
  - Network data
  - Memory values from the game
  - Anything that depends on internal game state

### Analysis Features

- Implements practical post-session analysis based only on mouse/input behavior.
- Calculates flick intensity, smoothness, micro-correction frequency, overflick/underflick tendency, stability/jitter, click timing patterns, session consistency, and trend over the last 10-20 sessions.

### Recommendations

- Provides cautious suggestions for aim adjustments based on analysis results.
- Generates recommendations for Universal, Tactical FPS, Hero/Arena FPS, Battle Royale, and specific games like CS2, Valorant, Apex Legends, and Overwatch 2.

# Aim/Mouse Behavior Analysis

## Overview

This is a Windows Python desktop app designed for analyzing aim/mouse behavior in games without affecting gameplay or reading game memory.

### Core Requirements

- **No Memory Reading**: The app does not read game memory.
- **No Process Injection**: It does not inject into the game process.
- **No Hooks**: It does not hook into the game.
- **No Input Emulation**: It does not emulate mouse/keyboard input.
- **No Packet Interception**: It does not intercept packets.
- **No Gameplay Interaction**: It runs as a completely separate process and only uses safe standard Windows APIs for passive input monitoring and active-window/process detection.

### Main Goal

The app is a local post-session analysis tool that records mouse behavior safely, detects the selected game or genre, stores session data, analyzes movement patterns, and shows recommendations after the session ends.

### Project Structure

- **app.py**: The main entry point of the application.
- **config.py**: Configuration settings for the app.
- **collector.py**: Collects mouse movement, clicks, timestamps, and active window/process information.
- **game_detector.py**: Detects the active game based on process name or active window title.
- **analyzer.py**: Analyzes mouse movement data to calculate flick intensity, smoothness, etc.
- **recommendations.py**: Generates recommendations based on analysis results.
- **storage.py**: Stores session data locally in JSON format.
- **ui/**: Contains the user interface files.
  - **main_window.py**: The main window of the application.
  - **game_selection.py**: Allows users to select a game or genre.
  - **dashboard.py**: Displays session statistics and recommendations.
  - **settings_panel.py**: Provides settings for the app.

### Game/Genre Selection

- Supports both genre mode and specific game mode.
- Genre options: Tactical FPS, Hero / Arena FPS, Battle Royale, Universal.
- Specific game options: CS2, Valorant, Apex Legends, Overwatch 2, Universal.
